  <label>Cadmium (Cd):</label><br>
  <input type="number" id="cd" placeholder="e.g., 0.8"><br>

  <label>Lead (Pb):</label><br>
  <input type="number" id="pb" placeholder="e.g., 2.5"><br>

  <button onclick="predict()">Predict</button>

  <p id="result"></p>

  <script>
    function predict() {
      const cd = parseFloat(document.getElementById("cd").value);
      const pb = parseFloat(document.getElementById("pb").value);
      let message = "";

      if (isNaN(cd) || isNaN(pb)) {
        message = "Please enter values for both metals.";
      } else {
        if (cd > 1 || pb > 3) {
          message = "⚠️ High heavy metal levels detected!";
        } else {
          message = "✅ Heavy metal levels are within safe limits.";
        }
      }

      document.getElementById("result").innerText = message;
    }
  </script>
