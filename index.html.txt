  <script>
    function predictHeavyMetals() {
      // Get input values
      const om = parseFloat(document.getElementById('organicMatter').value);
      const ph = parseFloat(document.getElementById('ph').value);
      const cec = parseFloat(document.getElementById('cec').value);

      if (isNaN(om) || isNaN(ph) || isNaN(cec)) {
        alert('Please enter valid numbers for all fields.');
        return;
      }

      // Dummy prediction formula (just for demo)
      // Replace with your own model or formula later!
      const cd = (om * 0.1 + ph * 0.05 + cec * 0.02).toFixed(2);
      const cr = (om * 0.2 + ph * 0.03 + cec * 0.04).toFixed(2);
      const cu = (om * 0.15 + ph * 0.07 + cec * 0.03).toFixed(2);
      const ni = (om * 0.12 + ph * 0.06 + cec * 0.025).toFixed(2);
      const pb = (om * 0.18 + ph * 0.04 + cec * 0.05).toFixed(2);
      const zn = (om * 0.25 + ph * 0.02 + cec * 0.01).toFixed(2);

      // Show results
      document.getElementById('cd').textContent = 'Cd: ' + cd;
      document.getElementById('cr').textContent = 'Cr: ' + cr;
      document.getElementById('cu').textContent = 'Cu: ' + cu;
      document.getElementById('ni').textContent = 'Ni: ' + ni;
      document.getElementById('pb').textContent = 'Pb: ' + pb;
      document.getElementById('zn').textContent = 'Zn: ' + zn;

      document.getElementById('results').style.display = 'block';
    }
  </script>
